import { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { api, setToken } from '../api';
import { Block, Kind, KIND_META, applyKind } from '../kinds';
import FlowCanvas from '../components/FlowCanvas';
import Inspector from '../components/Inspector';
import Subscribers from '../components/Subscribers';
import Broadcasts from '../components/Broadcasts';

const ADD_KINDS: Kind[] = ['message', 'command', 'text', 'input'];

export default function BotEditor() {
  const { botId } = useParams();
  const nav = useNavigate();
  const [bot, setBot] = useState<any>(null);
  const [blocks, setBlocks] = useState<Block[]>([]);
  const [selectedId, setSelectedId] = useState<string | null>(null);
  const [tab, setTab] = useState<'flow' | 'subs' | 'broadcasts'>('flow');
  const [toast, setToast] = useState<{ text: string; err?: boolean } | null>(null);
  const [loading, setLoading] = useState(true);

  function flash(text: string, err = false) {
    setToast({ text, err });
    setTimeout(() => setToast(null), 3000);
  }

  async function load() {
    try {
      const [b, bl] = await Promise.all([api.get(`/bots/${botId}`), api.get(`/bots/${botId}/blocks`)]);
      setBot(b);
      setBlocks(bl);
      if (!selectedId && bl.length) setSelectedId(bl[0].id);
    } catch {
      nav('/');
    } finally {
      setLoading(false);
    }
  }
  useEffect(() => { load(); }, [botId]);

  async function toggleActive() {
    try {
      const action = bot.isActive ? 'deactivate' : 'activate';
      const updated = await api.post(`/bots/${botId}/${action}`);
      setBot(updated);
      flash(updated.isActive ? 'Бот запущен ✓' : 'Бот остановлен');
    } catch (e: any) {
      flash(e.message, true);
    }
  }

  async function saveBlock(draft: Block) {
    try {
      await api.patch(`/blocks/${draft.id}`, {
        name: draft.name, type: draft.type, triggerType: draft.triggerType,
        triggerValue: draft.triggerValue, text: draft.text, mediaType: draft.mediaType,
        mediaUrl: draft.mediaUrl, variableName: draft.variableName, nextBlockId: draft.nextBlockId,
      });
      const withButtons = await api.put(`/blocks/${draft.id}/buttons`, { buttons: draft.buttons });
      setBlocks((prev) => prev.map((b) => (b.id === draft.id ? withButtons : b)));
      flash('Сохранено ✓');
    } catch (e: any) {
      flash(e.message, true);
    }
  }

  async function moveBlock(id: string, x: number, y: number) {
    setBlocks((prev) => prev.map((b) => (b.id === id ? { ...b, posX: x, posY: y } : b)));
    api.patch(`/blocks/${id}`, { posX: x, posY: y }).catch(() => {});
  }

  async function addBlock(kind: Kind) {
    const patch = applyKind(kind);
    const created = await api.post(`/bots/${botId}/blocks`, {
      ...patch,
      name: KIND_META[kind].label,
      posX: 140 + (blocks.length % 6) * 34,
      posY: 140 + (blocks.length % 6) * 34,
    });
    setBlocks((prev) => [...prev, created]);
    setSelectedId(created.id);
  }

  async function deleteBlock(id: string) {
    await api.del(`/blocks/${id}`);
    setBlocks((prev) => prev.filter((b) => b.id !== id));
    setSelectedId(null);
  }

  if (loading) return <div className="app"><div className="center"><div className="spinner" /></div></div>;

  const selected = blocks.find((b) => b.id === selectedId) || null;

  return (
    <div className="app">
      <div className="topbar">
        <button className="btn-ghost topbar-link" onClick={() => nav('/')}>← Боты</button>
        <div className="brand" style={{ fontSize: 15 }}>{bot.name}</div>
        {bot.username && <span className="card-user">@{bot.username}</span>}
        <span className={`status-pill ${bot.isActive ? 'status-on' : 'status-off'}`}>
          <span className="status-dot" /> {bot.isActive ? 'Работает' : 'Остановлен'}
        </span>
        <div className="spacer" />
        <div className="tabs">
          <button className={`tab ${tab === 'flow' ? 'active' : ''}`} onClick={() => setTab('flow')}>Сценарий</button>
          <button className={`tab ${tab === 'subs' ? 'active' : ''}`} onClick={() => setTab('subs')}>Подписчики</button>
          <button className={`tab ${tab === 'broadcasts' ? 'active' : ''}`} onClick={() => setTab('broadcasts')}>Рассылки</button>
        </div>
        <button className={`btn ${bot.isActive ? '' : 'btn-primary'}`} onClick={toggleActive}>
          {bot.isActive ? 'Остановить' : 'Запустить'}
        </button>
        <button className="topbar-link" onClick={() => { setToken(null); location.href = '/login'; }}>Выйти</button>
      </div>

      {tab === 'flow' && (
        <div className="editor">
          <div style={{ position: 'relative', flex: 1, minWidth: 0, display: 'flex' }}>
            <div className="canvas-toolbar">
              {ADD_KINDS.map((k) => (
                <button key={k} className="btn btn-sm" onClick={() => addBlock(k)}>
                  <span className={`node-type-dot ${KIND_META[k].cls}`} style={{ display: 'inline-block', marginRight: 6 }} />
                  {KIND_META[k].label}
                </button>
              ))}
            </div>
            <FlowCanvas blocks={blocks} selectedId={selectedId} onSelect={setSelectedId} onMove={moveBlock} />
          </div>
          {selected ? (
            <Inspector block={selected} blocks={blocks} onSave={saveBlock} onDelete={deleteBlock} />
          ) : (
            <div className="inspector"><div className="inspector-inner"><p className="hint">Выберите блок на канве или добавьте новый.</p></div></div>
          )}
        </div>
      )}

      {tab === 'subs' && <Subscribers botId={botId!} />}
      {tab === 'broadcasts' && <Broadcasts botId={botId!} />}

      {toast && <div className={`toast ${toast.err ? 'err' : ''}`}>{toast.text}</div>}
    </div>
  );
}
